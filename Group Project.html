rs = RandomOverSampler(random_state=0); X_resampled, y_resampled = rs.fit_resample(X, y)
X_train, X_test, y_train, y_test = train_test_split(X_resampled, y_resampled, test_size=.2, random_state=0, 
                                                    stratify=y_resampled)
model = GradientBoostingClassifier(n_estimators=100, learning_rate=0.25, max_depth=1, random_state=0)
model.fit(X_train, y_train); y_hat = model.predict(X_test)
print("Here is the data of gradient boosting classifier after oversampling.")
print(f'The accuracy is {accuracy_score(y_true=y_test, y_pred=model.predict(X_test)):.3}')
print(f'The precision is {precision_score(y_true=y_test, y_pred=model.predict(X_test)):.3}')
print(f'The recall is {recall_score(y_true=y_test, y_pred=model.predict(X_test)):.3}')
print(f"The AUC is {roc_auc_score(y_true=y_test, y_score=model.predict(X_test)):.3}")

CM = confusion_matrix(y_true=y_test, y_pred=model.predict(X_test)); GBC_correct_predictions = CM[0][0] + CM[1][1]
print(f"There are {GBC_correct_predictions} correct predictions in total")
print(f"There are {CM[1][1]} correct predictions for the dead")
print(f"There are {CM[0][0]} correct predictions for the alive")
print(f"There are {CM[1][0]} wrong predictions for the dead")
print(f"There are {CM[0][1]} wrong predictions for the alive")
axis = [item for item in range(0, 2)]
DF_CM = pd.DataFrame(CM, index = [i for i in axis], columns = [i for i in axis]); plt.figure(figsize=(7, 5))
HM = plt.axes(); sn.heatmap(DF_CM, annot=True)
